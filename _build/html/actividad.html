Lanzamiento de Objeto#

Supuestos del Modelo#

Para modelar este problema de forma analítica, consideraremos los siguientes supuestos físicos:

  • Aceleración constante en el empuje: Asumimos que la mano aplica una fuerza constante sobre el objeto, lo que genera una aceleración constante (\(a_{\text{empuje}}\)) hacia arriba durante el trayecto \(h_1\).

  • Resistencia del aire nula: Despreciamos los efectos de la fricción con el aire.

  • Aceleración de gravedad (\(g\)): Una vez que el objeto abandona la mano, la única aceleración que actúa sobre él es la de gravedad (\(g \approx 9.81\) m/s²), dirigida hacia abajo.

 

 

Gráficos de velocidad y aceleración#

A continuación, se representan las curvas de velocidad y aceleración en función del tiempo para ambas fases del movimiento. Se puede observar geométricamente cómo el área bajo la curva en el gráfico de velocidad corresponde a los desplazamientos \(h_1\) (fase de empuje) y \(h_2\) (vuelo libre).

 

 

Interpretación de las gráficas: áreas y pendientes#

Al graficar el movimiento, podemos observar propiedades fundamentales de la cinemática:

  • Gráfica de Velocidad vs. Tiempo (\(v-t\)):

    • Pendientes: La pendiente de esta gráfica representa la aceleración. Durante la Fase 1, la pendiente es positiva y constante (el valor es \(a_{\text{empuje}}\)). Durante la Fase 2, la pendiente es negativa y constante (el valor es \(-g\)).

    • Áreas: El área bajo la curva de velocidad representa el desplazamiento (\(\Delta y\)). El área del primer triángulo formado (base \(t_1\), altura \(v_1\)) es exactamente igual a \(h_1\). El área del segundo triángulo (base \(t_2\), altura \(v_1\)) es exactamente \(h_2\).

  • Gráfica de Aceleración vs. Tiempo (\(a-t\)):

    • Áreas: El área bajo la curva de aceleración representa el cambio de velocidad (\(\Delta v\)). El área del rectángulo positivo correspondiente a la Fase 1 es igual a \(v_1\) (la rapidez que gana). El área del rectángulo negativo de la Fase 2 es \(-v_1\) (la rapidez que pierde hasta detenerse por completo).

Cálculo de áreas y pendientes#

Calculando las áreas bajo la curva en el gráfico de velocidad, es decir, el área de los triángulos obtenemos las siguientes relaciones:

\begin{aligned} h_1 &= \frac{1}{2} t_1 v_1 \ h_2 &= \frac{1}{2} (t_2 - t_1) v_1 \end{aligned}

Y calculando las pendientes en el mismo gráfico se obtiene:

\begin{aligned} a_{\text{empuje}} &= \frac{v_1}{t_1} \ g &= \frac{v_1}{t_2 - t_1} \end{aligned}

A partir de estas ecuaciones obtenemos los parámetros en función de las distancias \(h_1\) y \(h_2\):

\begin{aligned} v_1 &= \sqrt{2 g h_2} \ t_1 &= \frac{2 h_1}{\sqrt{2 g h_2}} \ t_2 &= \frac{2}{\sqrt{2 g h_2}} (h_1 + h_2) \ a_{\text{empuje}} &= \frac{h_2}{h_1} g \end{aligned}

<script>
    (function() {
        function initSimulation() {
            const canvasSim = document.getElementById('canvas_sim_UID');
            const canvasGraph = document.getElementById('canvas_graphs_UID');
            if (!canvasSim || !canvasGraph) {
                setTimeout(initSimulation, 50); 
                return;
            }
            
            const ctxSim = canvasSim.getContext('2d');
            const ctxG = canvasGraph.getContext('2d');
            
            const btn_play = document.getElementById('play_UID');
            const btn_reset = document.getElementById('reset_UID');
            const btn_update = document.getElementById('update_params_UID');
            const time_display = document.getElementById('time_display_UID');
            const speed_slider = document.getElementById('speed_UID');
            const speed_val = document.getElementById('speed_val_UID');
            const results_div = document.getElementById('results_UID');
            
            // Entradas
            const input_h1 = document.getElementById('h1_UID');
            const input_h2 = document.getElementById('h2_UID');

            let isPlaying = false;
            let animationId;
            let lastTime = 0; 
            
            // Físicas
            const g = 9.81;
            let h1, h2, v1, a_imp, t1, t2, tMax, hTotal;
            let t = 0;

            function calculatePhysics() {
                h1 = parseFloat(input_h1.value);
                h2 = parseFloat(input_h2.value);
                
                if(h1 <= 0) h1 = 0.1;
                if(h2 <= 0) h2 = 0.1;
                input_h1.value = h1;
                input_h2.value = h2;

                hTotal = h1 + h2;
                v1 = Math.sqrt(2 * g * h2);
                a_imp = g * (h2 / h1);
                t1 = h1 * Math.sqrt(2 / (g * h2));
                t2 = Math.sqrt(2 * h2 / g);
                tMax = t1 + t2;

                results_div.innerHTML = `
                    <strong>Resultados calculados:</strong><br>
                    • Rapidez de despegue (v₁): <b>${v1.toFixed(2)} m/s</b><br>
                    • Aceleración de empuje (a): <b>${a_imp.toFixed(2)} m/s²</b><br>
                    • Tiempo de empuje (t₁): <b>${t1.toFixed(3)} s</b> | Tiempo de vuelo libre (t₂): <b>${t2.toFixed(3)} s</b><br>
                    • Tiempo total a altura máxima: <b>${tMax.toFixed(3)} s</b>
                `;
            }

            function getState(currentTime) {
                if (currentTime <= t1) {
                    // Fase 1: Empuje
                    let currentY = 0.5 * a_imp * currentTime * currentTime;
                    let currentV = a_imp * currentTime;
                    return { y: currentY, v: currentV, a: a_imp, phase: 1 };
                } else if (currentTime <= tMax) {
                    // Fase 2: Vuelo libre
                    let dt = currentTime - t1;
                    let currentY = h1 + (v1 * dt) - (0.5 * g * dt * dt);
                    let currentV = v1 - g * dt;
                    return { y: currentY, v: currentV, a: -g, phase: 2 };
                } else {
                    // Fin (Altura máxima alcanzada)
                    return { y: hTotal, v: 0, a: -g, phase: 3 };
                }
            }

            speed_slider.addEventListener('input', () => {
                speed_val.innerText = parseFloat(speed_slider.value).toFixed(1) + 'x';
            });
            
            function drawSim() {
                ctxSim.clearRect(0, 0, canvasSim.width, canvasSim.height);
                let state = getState(t);
                
                // Márgenes y escala
                let marginY = 30;
                let pxPerMeter = (canvasSim.height - 2 * marginY) / (hTotal * 1.2); 
                let groundY = canvasSim.height - marginY;

                // Dibujar Suelo/Origen
                ctxSim.fillStyle = "#adb5bd";
                ctxSim.fillRect(0, groundY, canvasSim.width, marginY);
                ctxSim.fillStyle = "#000";
                ctxSim.font = "12px Arial";
                ctxSim.fillText("y = 0", 10, groundY + 15);

                // Dibujar marcas de h1 y altura máxima
                let h1Px = groundY - (h1 * pxPerMeter);
                let hMaxPx = groundY - (hTotal * pxPerMeter);

                ctxSim.strokeStyle = "#0d6efd";
                ctxSim.setLineDash([5, 5]);
                ctxSim.beginPath(); ctxSim.moveTo(0, h1Px); ctxSim.lineTo(canvasSim.width, h1Px); ctxSim.stroke();
                ctxSim.fillText(`h₁ (${h1}m) - Despegue`, 10, h1Px - 5);

                ctxSim.strokeStyle = "#dc3545";
                ctxSim.beginPath(); ctxSim.moveTo(0, hMaxPx); ctxSim.lineTo(canvasSim.width, hMaxPx); ctxSim.stroke();
                ctxSim.fillText(`h₁ + h₂ (${hTotal}m) - Max`, 10, hMaxPx - 5);
                ctxSim.setLineDash([]);

                // Dibujar la "Mano" (solo sube hasta h1)
                let handY = (state.phase === 1) ? state.y : h1;
                let handYPx = groundY - (handY * pxPerMeter);
                ctxSim.fillStyle = "#ffc107"; 
                ctxSim.fillRect(canvasSim.width/2 - 25, handYPx, 50, 10);

                // Dibujar el Objeto (Bola)
                let ballYPx = groundY - (state.y * pxPerMeter);
                ctxSim.fillStyle = "#198754";
                ctxSim.beginPath();
                ctxSim.arc(canvasSim.width/2, ballYPx - 10, 10, 0, Math.PI*2);
                ctxSim.fill();
                ctxSim.stroke();
            }

            function drawGraphs() {
                ctxG.clearRect(0, 0, canvasGraph.width, canvasGraph.height);
                
                let margin = 40;
                let graphW = 180; 
                let graphH = 280; 
                
                let offY = 50;  
                let offV = 280; 
                let offA = 510; 
                
                let bottomY = margin + graphH;

                // --- Ejes Comunes ---
                ctxG.strokeStyle = "#000";
                ctxG.fillStyle = "#000";
                ctxG.font = "12px Arial";

                [
                    {x: offY, title: "Posición y(t) [m]"}, 
                    {x: offV, title: "Velocidad v(t) [m/s]"}, 
                    {x: offA, title: "Aceleración a(t) [m/s²]"}
                ].forEach(ax => {
                    ctxG.beginPath();
                    ctxG.moveTo(ax.x, margin); ctxG.lineTo(ax.x, bottomY); 
                    ctxG.lineTo(ax.x + graphW, bottomY); 
                    ctxG.stroke();
                    ctxG.fillText(ax.title, ax.x - 10, margin - 15);
                    ctxG.fillText("t (s)", ax.x + graphW - 20, bottomY + 20);
                });

                // Escalas
                let scaleT = graphW / (tMax * 1.1); 
                let scalePos = graphH / (hTotal * 1.2); 
                let scaleVel = graphH / (v1 * 1.5);  
                
                let maxA = Math.max(a_imp, g);
                let scaleAcc = (graphH / 2) / (maxA * 1.2); 
                let midA_Y = margin + graphH / 2; 
                
                ctxG.strokeStyle = "#ccc";
                ctxG.beginPath(); ctxG.moveTo(offA, midA_Y); ctxG.lineTo(offA + graphW, midA_Y); ctxG.stroke();
                ctxG.fillText("0", offA - 15, midA_Y + 5);

                ctxG.lineWidth = 2.5;
                
                // Dibujar Curvas
                ctxG.beginPath(); ctxG.strokeStyle = "#198754"; 
                for(let i=0; i<=t; i+=0.01) {
                    let s = getState(i);
                    let px = offY + i * scaleT;
                    let py = bottomY - s.y * scalePos;
                    i===0 ? ctxG.moveTo(px, py) : ctxG.lineTo(px, py);
                } ctxG.stroke();
                
                ctxG.beginPath(); ctxG.strokeStyle = "#0d6efd"; 
                for(let i=0; i<=t; i+=0.01) {
                    let s = getState(i);
                    let px = offV + i * scaleT;
                    let py = bottomY - s.v * scaleVel;
                    i===0 ? ctxG.moveTo(px, py) : ctxG.lineTo(px, py);
                } ctxG.stroke();

                ctxG.beginPath(); ctxG.strokeStyle = "#dc3545"; 
                for(let i=0; i<=t; i+=0.005) {
                    let s = getState(i);
                    let px = offA + i * scaleT;
                    let py = midA_Y - s.a * scaleAcc;
                    
                    if (i === 0) {
                        ctxG.moveTo(px, py);
                    } else {
                        if (Math.abs(i - t1) < 0.01) {
                            ctxG.lineTo(px, midA_Y - (-g) * scaleAcc);
                        } else {
                            ctxG.lineTo(px, py);
                        }
                    }
                } ctxG.stroke();

                if (t >= t1) {
                    let pxT1 = t1 * scaleT;
                    ctxG.fillStyle = "rgba(0, 0, 0, 0.4)";
                    ctxG.setLineDash([3, 3]);
                    ctxG.beginPath(); ctxG.moveTo(offY + pxT1, margin); ctxG.lineTo(offY + pxT1, bottomY); ctxG.stroke();
                    ctxG.beginPath(); ctxG.moveTo(offV + pxT1, margin); ctxG.lineTo(offV + pxT1, bottomY); ctxG.stroke();
                    ctxG.beginPath(); ctxG.moveTo(offA + pxT1, margin); ctxG.lineTo(offA + pxT1, bottomY); ctxG.stroke();
                    ctxG.setLineDash([]);
                }
            }

            function animate(timestamp) {
                if (!isPlaying) return;
                if (!lastTime) lastTime = timestamp;
                
                let deltaTime = (timestamp - lastTime) / 1000;
                lastTime = timestamp;
                
                let speed_multiplier = parseFloat(speed_slider.value);
                t += deltaTime * speed_multiplier;
                
                if (t >= tMax) {
                    t = tMax;
                    isPlaying = false;
                    btn_play.innerText = "▶ Reproducir / Pausar";
                }
                
                time_display.innerText = t.toFixed(2);
                drawSim();
                drawGraphs();
                
                if(isPlaying) {
                    animationId = requestAnimationFrame(animate);
                }
            }

            btn_update.addEventListener('click', () => {
                isPlaying = false;
                t = 0;
                lastTime = 0;
                time_display.innerText = "0.00";
                calculatePhysics();
                drawSim();
                drawGraphs();
                btn_play.innerText = "▶ Reproducir / Pausar";
            });

            btn_play.addEventListener('click', () => {
                isPlaying = !isPlaying;
                if (t >= tMax) {
                    t = 0; 
                }
                if (isPlaying) {
                    lastTime = 0; 
                    animationId = requestAnimationFrame(animate);
                }
            });

            btn_reset.addEventListener('click', () => {
                isPlaying = false;
                t = 0;
                lastTime = 0;
                time_display.innerText = "0.00";
                drawSim();
                drawGraphs();
            });
            
            calculatePhysics();
            drawSim();
            drawGraphs();
        }
        
        initSimulation();
    })();
</script>
"""
